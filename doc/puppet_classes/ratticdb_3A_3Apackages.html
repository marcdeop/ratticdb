class ratticdb::packages {
  include ratticdb::params
  case $::osfamily {
    'debian': {
      include apt
      exec { 'updatePackages':
        command => 'apt-get update',
        before  => Package['python'],
        path    => '/usr/bin/',
      }
    }
    'redhat': {
      include epel
      package { 'wget':
        ensure => installed,
      }
    }
  default: {
    }
  }

    ensure  => latest,
    require => Class[$::ratticdb::params::managing_class],
